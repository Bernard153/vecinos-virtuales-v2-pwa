📦 OPCIÓN 1: NETLIFY (MÁS RÁPIDO - 5 MINUTOS)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Ejecuta: CREAR_ZIP_DEPLOYMENT.bat
2. Ve a: https://app.netlify.com/drop
3. Arrastra el archivo ZIP
4. Espera 30 segundos
5. ¡Listo! Copia la URL

URL ejemplo: https://vecinos-virtuales.netlify.app


📤 OPCIÓN 2: FTP (HOSTING TRADICIONAL)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Archivos a subir:
✅ index.html
✅ terminos.html
✅ LIMPIAR_DATOS.html (opcional)
✅ .htaccess
✅ Carpeta css/ (completa)
✅ Carpeta js/ (completa)

Subir a: public_html/ o www/


🌐 OPCIÓN 3: VERCEL
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Instala: npm i -g vercel
2. En la carpeta del proyecto: vercel
3. Sigue instrucciones
4. ¡Listo!
